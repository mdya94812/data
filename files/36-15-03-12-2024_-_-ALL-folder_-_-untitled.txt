class Problem extends Thread {    public void run() {        try {            System.out.println("Thread is running...");            // Simulating some work            for (int i = 0; i < 5; i++) {                System.out.println(i);                Thread.sleep(1000); // Sleep for 1 second            }        } catch (InterruptedException e) {            System.out.println("Thread interrupted: " + e);        }    }}public class Main {    public static void main(String[] args) {        Problem t1 = new Problem();        t1.start(); // Starting the thread                // Ensure the main thread waits for the child thread to finish        try {            t1.join(); // Main thread waits for t1 to complete        } catch (InterruptedException e) {            System.out.println("Main thread interrupted: " + e);        }        System.out.println("Main thread execution completed.");    }}
